# Description: Create a reusable floating modal component that will display cards with a specific status (Done, Archived, or Deleted).
# Details:
Implement a modal component that can be triggered from the kebab menu. The modal should display a list of cards with the selected status, allow viewing card details, and provide options to restore cards. The modal should be dismissible and follow accessibility best practices. Example implementation:
```jsx
const StatusModal = ({ status, isOpen, onClose }) => {
  const [cards, setCards] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  
  useEffect(() => {
    if (!isOpen) return;
    
    const fetchCards = async () => {
      setLoading(true);
      try {
        // Use the appropriate fetch function based on status
        const fetchFunction = {
          'done': fetchDoneCards,
          'archived': fetchArchivedCards,
          'deleted': fetchDeletedCards
        }[status];
        
        const boardId = getCurrentBoardId(); // Implement this function
        const fetchedCards = await fetchFunction(boardId);
        setCards(fetchedCards);
        setError(null);
      } catch (err) {
        setError('Failed to load cards');
        console.error(err);
      } finally {
        setLoading(false);
      }
    };
    
    fetchCards();
  }, [isOpen, status]);
  
  return (
    <div className={`modal ${isOpen ? 'open' : ''}`}>
      <div className="modal-content">
        <div className="modal-header">
          <h2>{status.charAt(0).toUpperCase() + status.slice(1)} Items</h2>
          <button onClick={onClose} aria-label="Close modal">Ã—</button>
        </div>
        
        <div className="modal-body">
          {loading && <p>Loading...</p>}
          {error && <p className="error">{error}</p>}
          {!loading && !error && cards.length === 0 && (
            <p>No {status} items found.</p>
          )}
          
          <ul className="card-list">
            {cards.map(card => (
              <li key={card.id} className="card-item">
                <div className="card-title">{card.title}</div>
                <button onClick={() => viewCardDetails(card)}>View</button>
                <button onClick={() => restoreCard(card)}>Restore</button>
              </li>
            ))}
          </ul>
        </div>
      </div>
    </div>
  );
};
```
