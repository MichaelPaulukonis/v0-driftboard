```typescript
describe('linkifyText', () => {
  it('should return empty string for empty input', () => {
    const result = linkifyText('');
    expect(result).toEqual('');
  });

  it('should return plain text for input without URLs', () => {
    const text = 'This is plain text without URLs';
    const result = linkifyText(text);
    expect(result).toEqual([text]);
  });

  it('should convert http URLs to clickable links', () => {
    const text = 'Check this link: http://example.com';
    const result = linkifyText(text);
    expect(result).toHaveLength(2);
    expect(result[0]).toEqual('Check this link: ');
    expect(result[1].props.href).toEqual('http://example.com');
  });

  // Additional tests for other URL patterns
});
```
