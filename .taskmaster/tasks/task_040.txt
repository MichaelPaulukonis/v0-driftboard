# Description: Create functionality to view the full details of a card when selected from the status-based modal views.
# Details:
Implement a detailed view of a card that appears when a user selects a card from the Done, Archived, or Deleted views. This should display all relevant card information in a readable format. Example implementation:
```jsx
const CardDetailView = ({ card, onClose, onRestore }) => {
  if (!card) return null;
  
  return (
    <div className="card-detail-container">
      <div className="card-detail-header">
        <h3>{card.title}</h3>
        <button onClick={onClose} aria-label="Close detail view">Ã—</button>
      </div>
      
      <div className="card-detail-content">
        <p>{card.content}</p>
        
        {/* Display other card properties like dates, tags, etc. */}
        {card.createdAt && (
          <div className="card-meta">
            <span>Created: {new Date(card.createdAt).toLocaleString()}</span>
          </div>
        )}
        
        {card.tags && card.tags.length > 0 && (
          <div className="card-tags">
            {card.tags.map(tag => (
              <span key={tag} className="tag">{tag}</span>
            ))}
          </div>
        )}
      </div>
      
      <div className="card-detail-actions">
        <button 
          className="restore-button" 
          onClick={() => onRestore(card)}
        >
          Restore to Board
        </button>
      </div>
    </div>
  );
};

// Usage within the StatusModal component
```
