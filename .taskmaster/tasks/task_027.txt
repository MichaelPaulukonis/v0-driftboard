# Status: done

# Description: Implement a reusable utility function that detects URLs in text and converts them to clickable links
# Details:
Create a new file `lib/utils.ts` with a `linkifyText()` function that:

1. Takes a string input and returns JSX
2. Uses regex to identify URLs with http/https protocols
3. Splits text into segments (regular text and URLs)
4. Returns JSX with URLs wrapped in `<a>` tags with appropriate attributes
5. Ensures links open in new tabs with `target="_blank"` and `rel="noopener noreferrer"`

Example implementation:
```typescript
import React from 'react';

export function linkifyText(text: string): React.ReactNode {
  if (!text) return '';
  
  // Basic URL regex for http/https protocols
  const urlRegex = /(https?:\/\/[^\s]+)/g;
  
  // Split text by URLs
  const parts = text.split(urlRegex);
  
  // Find all URLs in the text
  const urls = text.match(urlRegex) || [];
  
  // Combine parts and URLs into a single array of React nodes
  const result: React.ReactNode[] = [];
  
  parts.forEach((part, i) => {
    // Add the text part
    if (part) result.push(part);
    
    // Add the URL part (if there is one)
    if (urls[i]) {
      result.push(
        <a 
          key={i} 
          href={urls[i]} 
          target="_blank" 
          rel="noopener noreferrer"
          className="text-blue-600 hover:underline"
        >
          {urls[i]}
        </a>
      );
    }
  });
  
  return result;
}
```
